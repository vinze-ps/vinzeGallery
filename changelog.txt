29.03.2021

- simple initialize and use
- many posibilities of customization and settings
- fully responsive
- unlimited instances of galleries on page
- buttons / touch & drag / scroll support for navigation in slider
- zoom and move photos in slider
- optional default layout for galleries (require bootstrap)



vinzeGallery.init({
  mobileWidthBreakpoint: 500, number (positive)
  gallery: {
    data: `data-vinze-gallery`, string
    smoothLoad: true, boolean
    smoothLoadSpeed: 50, number (positive)
    photosContainerData: `data-vinze-photo-container`, string
    photosData: `data-vinze-photo`, string
	photosTransitionDuration: 300, number (positive)
    photosEasing: "easeOutSine", string (1 of 29 predefined easings)
    photosHover: "zoomIn", "zoomOut", string
    layout: {
	  xl: false,
	  lg: 3,
	  md: 4,
	  sm: false,
	  xs: 6,
    },
  }
  slider: {
	data: `data-vinze-slider`, string
	transitionDuration: 300, number (max 600, positive)
	navButtons: "desktop"/"mobile"/true/false, string/boolean
	escButton: true/false/"desktop"/"mobile", boolean/string
	navTouch: true/false/"mobile"/"desktop", boolean/string
	closeViaDrag: true/false/"mobile"/"desktop", boolean/string
	mouseWheel: "zoom"/"navigation"/false, string/boolean
	photosIndicator: "desktop"/"mobile"/true/false, string/boolean
	photosEasing: "easeOutSine", string (1 of 29 predefined easings)
	photosGap: 15, number (max 300)
	photoTransitionDuration: 300, number (max 600, positive)
	zoomTransitionDuration: 200, number (max 600, positive)
    zoomMaxScale: 3, number (positive)
    zoomSpeed: 0.25, number (min 0.1, positive)
	loop: true/false/"mobile"/"desktop", boolean/string
	draggingXBreakpoint: 15, number (positive)
	draggingYBreakpoint: 15, number (positive)
  },
});

// <img src="for-thumbnail/for-slider" data-vinze-slider-src="for-slider" /> --- custom source;
// data-vinze-gallery --- gallery container attribute
// data-vinze-photo --- photo attribute for <img /> tag
